double pow_10 (int power)
{
    double res = 1;

    if (power > 0)
    {
        for (int i = 0; i < power; i++)
        {
            res *= 10;
        }
    }

    else if (power < 0)
    {
        for (int i = 0; i > power; i--)
        {
            res /= 10;
        }
    }

    return res;
}

int Arg_Scan (char* ptrarg, char* code, size_t num_of_line, char cmd)
{
    switch (cmd)
    {
        case CMD_PUSH:
            if (*ptrarg == '[')
            {
                return Reg_Scan (ptrarg, code, num_of_line);
            }
            else if (*ptrarg >= '0' && *ptrarg <= '9')
            {
                return Double_Arg_Scan (ptrarg, code, num_of_line);
            }
            else
            {
                fprintf (logfile, "Incorrect argument of push in line %zd\n", num_of_line + 1);
                return ERR;
            }
            break;
        case CMD_POP:
            if (*ptrarg == '[')
            {
                return Reg_Scan (ptrarg, code, num_of_line);
            }
            else if (*ptrarg == '\n' || *ptrarg == '\0')
            {
                printf ("No arg\n");
                *code = NO_ARG;
                return NO_ERR;
            }
            else
            {
                fprintf (logfile, "Incorrect argument of pop in line %zd\n", num_of_line + 1);
                return ERR;
            }
            break;
        default:
            fprintf (logfile, "Command without argument, but has argument in commands.h \n");
            return ERR;
            break;
    }
}

int Reg_Scan (char* ptrarg, char* code, size_t num_of_line)
{
    char reg = 0;

    for (int i = 0; i < 4; i++)
    {
        if (strncmp (reg_name[i], ptrarg, 5) == 0)
        {
            reg = i + 1;
        }
    }
    if (reg == 0)
    {
        fprintf (logfile, "Incorrect register in line %zd", num_of_line + 1);
        return ERR;
    }
    else
    {
        *code = REGISTER;
        code++;
        *code = reg;
        return NO_ERR;
    }
}

int Double_Arg_Scan (char* ptrarg, char* code, size_t num_of_line)

        dot_pos++;
    }

    if (ptrarg[dot_pos] != '.' && ptrarg[dot_pos] && ptrarg[dot_pos] != ' ' && ptrarg[dot_pos] != '\t' && ptrarg[dot_pos] != '\n' && ptrarg[dot_pos] != '\0')
    {
        fprintf (logfile, "Fucking symbols in argument in line %zd\n", num_of_line + 1);
        return ERR;
    }
    else
    {
        while (*ptrarg != ' ' && *ptrarg != '\t' && *ptrarg != '\0' && *ptrarg != '\n')
        {
            if (*ptrarg != '.')
            {
                darg += (*ptrarg - '0') * pow_10 (dot_pos - 1);
                dot_pos--;
            }
            else
            {
                dot_counter++;
            }
            ptrarg++;
        }
        if (dot_counter > 1)

            return ERR;
        }
        *code = DOUBLE_ARG;
        code++;
        *((double*)code) = darg;
        return NO_ERR;
    }
